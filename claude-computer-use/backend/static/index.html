<html> <head> <title>Claude Computer Use Demo</title> </head> <body> <h1>Create Session</h1> <input id="api_key" type="text" placeholder="API Key"> <button onclick="createSession()">Create</button> <div id="chat"></div> <input id="message" type="text" placeholder="Type message..."> <button onclick="sendMessage()">Send</button> <iframe id="vnc" style="width:1024px; height:768px;" src=""></iframe> <script> let currentSession = null; let ws = null; async function createSession(){ const api_key = document.getElementById('api_key').value; const response = await fetch('/sessions', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({model: 'claude-sonnet-4-20250514', provider: 'anthropic', api_key: api_key}) }); const data = await response.json(); currentSession = data.session_id; connectWS(); loadMessages(); loadVNC(); } function connectWS(){ ws = new WebSocket(`ws://localhost:8000/ws/${currentSession}`); ws.onmessage = (event) => { const data = JSON.parse(event.data); const chat = document.getElementById('chat'); chat.innerHTML += `<p>${JSON.stringify(data)}</p>`; }; } async function loadMessages(){ const response = await fetch(`/sessions/${currentSession}/messages`); const messages = await response.json(); const chat = document.getElementById('chat'); chat.innerHTML = messages.map(m => `<p>${m.role}: ${JSON.stringify(m.content)}</p>`).join(''); } async function sendMessage(){ const msg = document.getElementById('message').value; await fetch(`/sessions/${currentSession}/messages`, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({message: msg}) }); document.getElementById('message').value = ''; } async function loadVNC(){ const response = await fetch(`/sessions/${currentSession}/vnc`); const data = await response.json(); document.getElementById('vnc').src = data.vnc_url; } </script> </body> </html>
